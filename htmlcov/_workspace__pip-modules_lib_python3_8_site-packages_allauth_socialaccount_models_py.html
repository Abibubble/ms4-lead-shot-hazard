151class SocialLogin(object): 

152 """ 

153 Represents a social user that is in the process of being logged 

154 in. This consists of the following information: 

155 

156 `account` (`SocialAccount` instance): The social account being 

157 logged in. Providers are not responsible for checking whether or 

158 not an account already exists or not. Therefore, a provider 

159 typically creates a new (unsaved) `SocialAccount` instance. The 

160 `User` instance pointed to by the account (`account.user`) may be 

161 prefilled by the provider for use as a starting point later on 

162 during the signup process. 

163 

164 `token` (`SocialToken` instance): An optional access token token 

165 that results from performing a successful authentication 

166 handshake. 

167 

168 `state` (`dict`): The state to be preserved during the 

169 authentication handshake. Note that this state may end up in the 

170 url -- do not put any secrets in here. It currently only contains 

171 the url to redirect to after login. 

172 

173 `email_addresses` (list of `EmailAddress`): Optional list of 

174 e-mail addresses retrieved from the provider. 

175 """ 

176 

177 def __init__(self, user=None, account=None, token=None, 

178 email_addresses=[]): 

179 if token: 

180 assert token.account is None or token.account == account 

181 self.token = token 

182 self.user = user 

183 self.account = account 

184 self.email_addresses = email_addresses 

185 self.state = {}

249 def lookup(self): 

250 """ 

251 Lookup existing account, if any. 

252 """ 

253 assert not self.is_existing 

254 try: 

255 a = SocialAccount.objects.get(provider=self.account.provider, 

256 uid=self.account.uid) 

257 # Update account 

258 a.extra_data = self.account.extra_data 

259 self.account = a 

260 self.user = self.account.user 

261 a.save() 

262 # Update token 

263 if app_settings.STORE_TOKENS and self.token and self.token.app.pk: 

264 assert not self.token.pk

266 t = SocialToken.objects.get(account=self.account, 

267 app=self.token.app) 

268 t.token = self.token.token 

269 if self.token.token_secret: 

270 # only update the refresh token if we got one 

271 # many oauth2 providers do not resend the refresh token 

272 t.token_secret = self.token.token_secret 

273 t.expires_at = self.token.expires_at 

274 t.save() 

275 self.token = t 

276 except SocialToken.DoesNotExist: 

277 self.token.account = a
