97 def render_mail(self, template_prefix, email, context): 

98 """ 

99 Renders an e-mail to `email`. `template_prefix` identifies the 

100 e-mail that is to be sent, e.g. "account/email/email_confirmation" 

101 """ 

102 subject = render_to_string('{0}_subject.txt'.format(template_prefix), 

103 context) 

104 # remove superfluous line breaks 

105 subject = " ".join(subject.splitlines()).strip() 

106 subject = self.format_email_subject(subject) 

107 

108 from_email = self.get_from_email() 

109 

110 bodies = {} 

111 for ext in ['html', 'txt']: 

112 try: 

113 template_name = '{0}_message.{1}'.format(template_prefix, ext) 

114 bodies[ext] = render_to_string(template_name, 

115 context).strip() 

116 except TemplateDoesNotExist: 

117 if ext == 'txt' and not bodies: 

118 # We need at least one body 

119 raise 

120 if 'txt' in bodies: 

121 msg = EmailMultiAlternatives(subject, 

122 bodies['txt'], 

123 from_email, 

124 [email]) 

125 if 'html' in bodies: 

126 msg.attach_alternative(bodies['html'], 'text/html') 

127 else: 

128 msg = EmailMessage(subject, 

129 bodies['html'], 

130 from_email, 

131 [email]) 

132 msg.content_subtype = 'html' # Main content is now text/html 

133 return msg

483 def authenticate(self, request, **credentials): 

484 """Only authenticates, does not actually login. See `login`""" 

485 from allauth.account.auth_backends import AuthenticationBackend 

486 

487 self.pre_authenticate(request, **credentials) 

488 AuthenticationBackend.unstash_authenticated_user() 

489 user = authenticate(request, **credentials) 

490 alt_user = AuthenticationBackend.unstash_authenticated_user() 

491 user = user or alt_user 

492 if user and app_settings.LOGIN_ATTEMPTS_LIMIT: 

493 cache_key = self._get_login_attempts_cache_key( 

494 request, **credentials) 

495 cache.delete(cache_key) 

496 else: 

497 self.authentication_failed(request, **credentials) 

498 return user
